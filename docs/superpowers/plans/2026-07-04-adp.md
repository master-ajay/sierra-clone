```sql
CREATE TABLE IF NOT EXISTS users (
    user_id       TEXT PRIMARY KEY,
    external_id   TEXT UNIQUE,
    display_name  TEXT NOT NULL,
    metadata      TEXT NOT NULL DEFAULT '{}',
    created_at    TEXT NOT NULL,
    updated_at    TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS sessions (
    session_id    TEXT PRIMARY KEY,
    user_id       TEXT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    status        TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'closed')),
    metadata      TEXT NOT NULL DEFAULT '{}',
    started_at    TEXT NOT NULL,
    updated_at    TEXT NOT NULL,
    closed_at     TEXT
);

CREATE TABLE IF NOT EXISTS messages (
    message_id    TEXT PRIMARY KEY,
    session_id    TEXT NOT NULL REFERENCES sessions(session_id) ON DELETE CASCADE,
    role          TEXT NOT NULL CHECK (role IN ('user', 'assistant', 'system')),
    content       TEXT NOT NULL,
    metadata      TEXT NOT NULL DEFAULT '{}',
    created_at    TEXT NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_sessions_user_id    ON sessions(user_id);
CREATE INDEX IF NOT EXISTS idx_sessions_status     ON sessions(status);
CREATE INDEX IF NOT EXISTS idx_messages_session_id ON messages(session_id);
CREATE INDEX IF NOT EXISTS idx_messages_created_at ON messages(created_at);
CREATE INDEX IF NOT EXISTS idx_users_external_id   ON users(external_id);
```

Note: `ON DELETE CASCADE` is added to both `sessions.user_id` and
`messages.session_id` so that `DELETE FROM users` removes all descendant rows
automatically. SQLite requires `PRAGMA foreign_keys = ON` per connection for
CASCADE to fire — this is set in `database.py`.

```python
import sqlite3
from pathlib import Path


def get_connection(db_path: str) -> sqlite3.Connection:
    conn = sqlite3.connect(db_path, check_same_thread=False)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA foreign_keys=ON")
    conn.commit()
    return conn


def run_migrations(db_path: str) -> None:
    Path(db_path).parent.mkdir(parents=True, exist_ok=True)
    migrations_dir = Path(__file__).parent.parent.parent.parent / "migrations"
    conn = get_connection(db_path)
    try:
        for sql_file in sorted(migrations_dir.glob("*.sql")):
            conn.executescript(sql_file.read_text())
        conn.commit()
    finally:
        conn.close()
```

Note: `run_migrations` resolves the `migrations/` directory relative to this
file's location (`src/adp/database.py` → four `parent` hops → repo subdirectory
root → `migrations/`). Adjust if the installed path differs; an alternative is
to pass the migrations path explicitly from `main.py`.

```python
from fastapi import Header, Depends
from adp.config import get_settings, Settings
from adp.errors import error_response


def require_api_key(
    x_api_key: str | None = Header(default=None),
    settings: Settings = Depends(get_settings),
) -> None:
    if x_api_key != settings.api_key:
        raise _auth_error()


def _auth_error():
    from fastapi import HTTPException
    from fastapi.responses import JSONResponse
    # We raise HTTPException so FastAPI converts it; the exception handler
    # in main.py rewrites the body to standard error shape.
    # Simpler: raise a custom exception handled in main.py.
    raise _UnauthorizedException()


class _UnauthorizedException(Exception):
    pass
```

Simpler implementation — put the logic directly in the dependency, avoid the
intermediate exception class:

```python
from fastapi import Header, Depends, Request
from fastapi.responses import JSONResponse
from adp.config import get_settings, Settings
from adp.errors import error_response


async def require_api_key(
    request: Request,
    x_api_key: str | None = Header(default=None),
    settings: Settings = Depends(get_settings),
) -> None:
    if x_api_key != settings.api_key:
        from fastapi import HTTPException
        raise HTTPException(
            status_code=401,
            detail={"code": "unauthorized", "message": "Invalid or missing API key", "details": {}},
        )
```

Use this second form. FastAPI's built-in `HTTPException` handler returns
`{"detail": ...}` by default; override it in `main.py` to return the standard
`{"error": ...}` shape.

```python
from contextlib import asynccontextmanager
from fastapi import FastAPI, Request
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError
from starlette.exceptions import HTTPException as StarletteHTTPException

from adp.config import get_settings
from adp.database import run_migrations
from adp.routes.system import router as system_router


@asynccontextmanager
async def lifespan(app: FastAPI):
    settings = get_settings()
    run_migrations(settings.db_path)
    yield


app = FastAPI(title="Agent Data Platform", version="1.0.0", lifespan=lifespan)


@app.exception_handler(StarletteHTTPException)
async def http_exception_handler(request: Request, exc: StarletteHTTPException):
    if isinstance(exc.detail, dict) and "code" in exc.detail:
        # Already in standard shape from auth dependency.
        return JSONResponse(
            status_code=exc.status_code,
            content={"error": exc.detail},
        )
    return JSONResponse(
        status_code=exc.status_code,
        content={"error": {"code": str(exc.status_code), "message": str(exc.detail), "details": {}}},
    )


@app.exception_handler(RequestValidationError)
async def validation_error_handler(request: Request, exc: RequestValidationError):
    return JSONResponse(
        status_code=422,
        content={"error": {"code": "validation_error", "message": "Request validation failed", "details": {"errors": exc.errors()}}},
    )


app.include_router(system_router)
```

- [ ] **Step 3: Write `src/adp/services/user_service.py`**

```python
import json
import sqlite3
from datetime import datetime, timezone
from uuid import uuid4

from adp.models.user import UserCreate, UserResponse, UserUpdate


def _now() -> str:
    return datetime.now(timezone.utc).isoformat()


def _row_to_user(row: sqlite3.Row) -> UserResponse:
    return UserResponse(
        user_id=row["user_id"],
        external_id=row["external_id"],
        display_name=row["display_name"],
        metadata=json.loads(row["metadata"]),
        created_at=row["created_at"],
        updated_at=row["updated_at"],
    )


def create_user(conn: sqlite3.Connection, data: UserCreate) -> UserResponse:
    user_id = str(uuid4())
    now = _now()
    try:
        conn.execute(
            """
            INSERT INTO users (user_id, external_id, display_name, metadata, created_at, updated_at)
            VALUES (?, ?, ?, ?, ?, ?)
            """,
            (user_id, data.external_id, data.display_name, json.dumps(data.metadata), now, now),
        )
        conn.commit()
    except sqlite3.IntegrityError:
        raise ValueError("duplicate_external_id")
    row = conn.execute("SELECT * FROM users WHERE user_id = ?", (user_id,)).fetchone()
    return _row_to_user(row)


def get_user(conn: sqlite3.Connection, user_id: str) -> UserResponse | None:
    row = conn.execute("SELECT * FROM users WHERE user_id = ?", (user_id,)).fetchone()
    return _row_to_user(row) if row else None


def get_user_by_external_id(conn: sqlite3.Connection, external_id: str) -> UserResponse | None:
    row = conn.execute("SELECT * FROM users WHERE external_id = ?", (external_id,)).fetchone()
    return _row_to_user(row) if row else None


def list_users(
    conn: sqlite3.Connection, limit: int = 20, cursor: str | None = None
) -> tuple[list[UserResponse], str | None]:
    if cursor:
        rows = conn.execute(
            "SELECT * FROM users WHERE created_at < ? ORDER BY created_at DESC LIMIT ?",
            (cursor, limit + 1),
        ).fetchall()
    else:
        rows = conn.execute(
            "SELECT * FROM users ORDER BY created_at DESC LIMIT ?", (limit + 1,)
        ).fetchall()

    has_more = len(rows) > limit
    items = [_row_to_user(r) for r in rows[:limit]]
    next_cursor = items[-1].created_at if has_more else None
    return items, next_cursor


def update_user(
    conn: sqlite3.Connection, user_id: str, data: UserUpdate
) -> UserResponse | None:
    existing = get_user(conn, user_id)
    if existing is None:
        return None
    new_name = data.display_name if data.display_name is not None else existing.display_name
```

```python
import json
import sqlite3
from datetime import datetime, timezone
from uuid import uuid4

from adp.models.session import SessionCreate, SessionResponse, SessionUpdate


def _now() -> str:
    return datetime.now(timezone.utc).isoformat()


def _row_to_session(row: sqlite3.Row) -> SessionResponse:
    return SessionResponse(
        session_id=row["session_id"],
        user_id=row["user_id"],
        status=row["status"],
        metadata=json.loads(row["metadata"]),
        started_at=row["started_at"],
        updated_at=row["updated_at"],
        closed_at=row["closed_at"],
    )


def create_session(
    conn: sqlite3.Connection, user_id: str, data: SessionCreate
) -> SessionResponse:
    session_id = str(uuid4())
    now = _now()
    conn.execute(
        """
        INSERT INTO sessions (session_id, user_id, status, metadata, started_at, updated_at, closed_at)
        VALUES (?, ?, 'active', ?, ?, ?, NULL)
        """,
        (session_id, user_id, json.dumps(data.metadata), now, now),
    )
    conn.commit()
    row = conn.execute(
        "SELECT * FROM sessions WHERE session_id = ?", (session_id,)
    ).fetchone()
    return _row_to_session(row)


def get_session(conn: sqlite3.Connection, session_id: str) -> SessionResponse | None:
    row = conn.execute(
        "SELECT * FROM sessions WHERE session_id = ?", (session_id,)
    ).fetchone()
    return _row_to_session(row) if row else None


def list_sessions(
    conn: sqlite3.Connection,
    user_id: str,
    status: str | None = None,
    limit: int = 20,
    cursor: str | None = None,
) -> tuple[list[SessionResponse], str | None]:
    base = "SELECT * FROM sessions WHERE user_id = ?"
    params: list = [user_id]
    if status:
        base += " AND status = ?"
        params.append(status)
    if cursor:
        base += " AND started_at < ?"
        params.append(cursor)
    base += " ORDER BY started_at DESC LIMIT ?"
    params.append(limit + 1)

    rows = conn.execute(base, params).fetchall()
    has_more = len(rows) > limit
    items = [_row_to_session(r) for r in rows[:limit]]
    next_cursor = items[-1].started_at if has_more else None
    return items, next_cursor


def update_session(
    conn: sqlite3.Connection, session_id: str, data: SessionUpdate
) -> SessionResponse | None:
    existing = get_session(conn, session_id)
    if existing is None:
        return None
    new_meta = data.metadata if data.metadata is not None else existing.metadata
    now = _now()
    conn.execute(
        "UPDATE sessions SET metadata = ?, updated_at = ? WHERE session_id = ?",
        (json.dumps(new_meta), now, session_id),
    )
    conn.commit()
    return get_session(conn, session_id)


def close_session(conn: sqlite3.Connection, session_id: str) -> SessionResponse | None:
    existing = get_session(conn, session_id)
    if existing is None:
        return None
    if existing.status == "closed":
        raise ValueError("already_closed")
    now = _now()
    conn.execute(
        "UPDATE sessions SET status = 'closed', closed_at = ?, updated_at = ? WHERE session_id = ?",
        (now, now, session_id),
    )
    conn.commit()
    return get_session(conn, session_id)
```

```python
import sqlite3

from adp.models.context import ContextRequest, ContextResponse, SessionSummary
from adp.models.message import MessageResponse
from adp.services import message_service, session_service, user_service


def assemble_context(conn: sqlite3.Connection, req: ContextRequest) -> ContextResponse | None:
    user = user_service.get_user(conn, req.user_id) if req.include_user_profile else None

    # Verify user exists regardless of include_user_profile.
    if user_service.get_user(conn, req.user_id) is None:
        return None

    # Session summary: count of all sessions and last interaction.
    all_sessions, _ = session_service.list_sessions(conn, req.user_id, limit=1000)
    session_count = len(all_sessions)
    last_interaction_at = all_sessions[0].updated_at if all_sessions else None
    summary = SessionSummary(session_count=session_count, last_interaction_at=last_interaction_at)

    messages: list[MessageResponse] = []
    if req.include_history and req.session_id:
        # Fetch up to max_messages most recent messages, then apply token budget.
        all_msgs, _ = message_service.list_messages(
            conn, req.session_id, limit=req.max_messages
        )
        messages = _apply_token_budget(all_msgs, req.max_tokens)

    token_estimate = _count_tokens(messages)

    return ContextResponse(
        user=user if req.include_user_profile else None,
        messages=messages,
        session_summary=summary,
        token_estimate=token_estimate,
    )


def _count_tokens(messages: list[MessageResponse]) -> int:
    return sum(len(m.content) // 4 for m in messages)


def _apply_token_budget(
    messages: list[MessageResponse], max_tokens: int
) -> list[MessageResponse]:
    # Messages arrive in chronological order (oldest first).
    # Truncate from the oldest until the budget is satisfied.
    while messages and _count_tokens(messages) > max_tokens:
        messages = messages[1:]
    return messages
```

```python
import sqlite3

from adp.models.message import MessageResponse
from adp.services.message_service import _row_to_message


def search_messages(
    conn: sqlite3.Connection,
    user_id: str,
    query: str,
    limit: int = 20,
    cursor: str | None = None,
) -> tuple[list[MessageResponse], str | None]:
    pattern = f"%{query}%"
    base = """
        SELECT m.*
        FROM messages m
        JOIN sessions s ON s.session_id = m.session_id
        WHERE s.user_id = ?
          AND m.content LIKE ? COLLATE NOCASE
    """
    params: list = [user_id, pattern]
    if cursor:
        base += " AND m.created_at > ?"
        params.append(cursor)
    base += " ORDER BY m.created_at ASC LIMIT ?"
    params.append(limit + 1)

    rows = conn.execute(base, params).fetchall()
    has_more = len(rows) > limit
    items = [_row_to_message(r) for r in rows[:limit]]
    next_cursor = items[-1].created_at if has_more else None
    return items, next_cursor
```
